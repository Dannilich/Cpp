#include <iostream>

using namespace std;


int main()
{
	setlocale(LC_ALL ,"ru");
	

	for (int k = 0; k < 10; k++)
	{
		for (int l = 0; l < 10; l++)
		{
			for (int m = 0; m < 10; m++)
			{
				for (int n = 0; n < 10; n++)
				{
					if ((k*10 + l) - (m*10 + n) == k + l + m + n)
					{
						cout
								<< "число: " << k*1000 + l*100 + m*10 + n

								<< "  k = " << k
								<< "  l = " << l
								<< "  m = " << m
								<< "  n = " << n
								
								<< endl
						;
					}
				}
			}
		}
	}

}
